papal = {
	trigger = {
		has_title = title:k_papal_state
	}
}

## CFP
norman = {
	trigger = {
		has_graphical_norman_culture_group_trigger = yes
		NOT = { has_title = title:k_papal_state }
	}
}

goidelic = {
	trigger = {
		has_graphical_goidelic_culture_group_trigger = yes
		NOT = { has_title = title:k_papal_state }
	}
}

andalusian = {
	trigger = {
		has_graphical_andalusian_culture_group_trigger = yes
		NOT = { has_title = title:k_papal_state }
	}
}
## Redefine mediterranean to exclude andalusians/popes from it
mediterranean = {
	trigger = {
		NOT = { has_title = title:k_papal_state }
		exists = capital_county
		capital_county.culture = {
			has_graphical_mediterranean_culture_group_trigger = yes
		}
		NOR = { 
			capital_county = {
				this = title:c_roma
			}
			has_graphical_andalusian_culture_group_trigger = yes
			has_title = title:k_papal_state
		}
	}
}
## Redefine western to exclude normans/goidelics/popes from it
western = {
	trigger = {
		exists = capital_county
		capital_county = {
			culture = {
				OR = {
					has_graphical_western_culture_group_trigger = yes
					has_graphical_norse_culture_group_trigger = yes
				}
			}
		}
		NOR = { 
			has_graphical_norman_culture_group_trigger = yes
			has_graphical_goidelic_culture_group_trigger = yes
			has_title = title:k_papal_state
		}
	}
}
