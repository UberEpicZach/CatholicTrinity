tct_neutral_struggle.0001 = {
	hidden = yes
	scope = struggle

	immediate = {
		title:e_hre.holder = {
			send_interface_toast = {
				type = event_toast_effect_bad
				title = tct_neutral_struggle.0001
				left_icon = ROOT
			}
		}
		## # Yearly drift every year that the Pope and Emperor dislike or like eachother
		investiture_hre_papal_opinion_effect = yes

		# Finally, queue the event up again for a year hence.
		trigger_event = {
			id = tct_neutral_struggle.0001
			years = 1
		}
	}
}

		reference = farmland
	}

		character = root
		animation = thinking
	}
	right_portrait = {
		character = title:d_angria.holder
		animation = anger
	}

	immediate = {
		culture:franconian = {
			change_cultural_acceptance = {
				target = culture:saxon
				value = -5
				desc = cultural_acceptance_historical_relations
			}
		}
		title:d_angria.holder = {
			add_opinion = {
				target = title:e_hre.holder
				modifier = insulted_opinion
				opinion = -50
			}
		}
		hidden_effect = {
			title:d_bavaria.holder = {
				add_opinion = {
					target = title:e_hre.holder
					modifier = insulted_opinion
					opinion = -50
				}
			}
			add_legitimacy = super_massive_legitimacy_loss
		}
	}

	option = {
		name = tct_neutral_struggle.0002.a

		title:d_angria.holder = {
			trigger_event = tct_neutral_struggle.0003
		}
	}
}
