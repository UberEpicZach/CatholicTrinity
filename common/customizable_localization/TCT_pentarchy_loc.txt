TCT_get_patriarchate_of_barony = {
	type = province
	text = {
		trigger = {
			NOT = {
				county.holder.top_liege.faith = {
					has_doctrine = doctrine_autocephaly_head
				}
				county.faith = {
					has_doctrine = doctrine_autocephaly_head
				}
			}
		}
		localization_key = TCT_NO_PATRIARCHATE_TITLE
		#fallback = yes
	}
	text = {
		localization_key = TCT_PATRIARCHATE_TITLE
		#fallback = yes
	}
}

TCT_get_InvestitureLaw_of_barony = {
	type = province
	text = {
		trigger = {
			county.holder.top_liege = {
				OR = {
					has_realm_law = free_investiture_early
					has_realm_law = free_investiture_late 
				}
			}
		}
		localization_key = TCT_FreeInvestitureMapMode
		#fallback = yes
	}
	text = {
		trigger = {
			county.holder.top_liege = {
				OR = {
					has_realm_law = papal_investiture_early
					has_realm_law = papal_investiture_late
				}
			}
		}
		localization_key = TCT_PapalInvestitureMapMode
		#fallback = yes
	}
}
