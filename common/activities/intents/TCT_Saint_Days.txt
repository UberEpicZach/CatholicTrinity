TCT_saint_days_reverence_intent = {
	icon = pious_intent

	auto_complete = yes

	is_shown = {

	}

	on_invalidated = {
		trigger_event = activity_system.0090
	}

	ai_will_do = {
		value = 1
		add = ai_zeal
		if = {
			limit = {
				ai_boldness > 0
			}
			add = {
				value = ai_boldness
				divide = 2
			}
		}
		if = {
			limit = {
				has_trait = zealous
			}
			add = 50
		}
		if = {
			limit = {
				OR = {
					has_trait = diligent
					has_trait = ambitious
				}
			}
			add = 10
		}
		if = {
			limit = {
				OR = {
					has_trait = shy
					has_trait = humble
					has_trait = reclusive
				}
			}
			subtract = 10
		}
	}

	scripted_animation = {
		animation = personality_honorable
	}
}
