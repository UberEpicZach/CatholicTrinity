rot_is_pope_of_target_trigger = {
	#save_temporary_scope_as = cardinal_check
	trigger_if = {
		limit = {
			scope:actor.faith = {
				NOT = { 
					has_variable_list = rot_antipopes 
				}
			} 
		}
		$TARGET$.faith.religious_head = $POPE$
	}
	trigger_else_if = {
		limit = {
			$TARGET$ = { 
				OR = {
					has_variable = rot_antipope_of_ruler
					AND = {
						exists = liege
						any_liege_or_above = {
							has_variable = rot_antipope_of_ruler
						}
					}
				}
			}
		} 
		OR = {
			AND = {
				$TARGET$ = {
					has_variable = rot_antipope_of_ruler
					var:rot_antipope_of_ruler = $POPE$
				}
			}
			$TARGET$ = {
				any_liege_or_above = {
					has_variable = rot_antipope_of_ruler
					var:rot_antipope_of_ruler = $POPE$
				}
			}
		}
	}
	trigger_else = {
		$TARGET$.faith.religious_head = $POPE$
	}
}

can_be_knight_trigger = {
	is_alive = yes
	is_adult = yes
	can_be_warrior_trigger = { ARMY_OWNER = $ARMY_OWNER$ }
	NOR = {
		has_trait = incapable
		has_trait = blind
		AND = {
			has_trait = devoted
			OR = {
				faith = {
					NOT = {
						has_doctrine_parameter = clergy_can_fight
					}
				}
				culture = {
					NOT = {
						has_cultural_parameter = culture_clergy_can_fight
					}
				}
			}
		}
		AND = {
			is_clergy = yes
			OR = {
				faith = {
					NOT = {
						has_doctrine_parameter = clergy_can_fight
					}
				}
			}
		}
	}

	trigger_if = {
		limit = {
			$ARMY_OWNER$.primary_title = {
				this = title:k_papal_state
			}
			NOR = {
				has_trait = incapable
				has_trait = blind
			}
		}
		prowess >= 5
	}
	
	trigger_if = {
		limit = {
			$ARMY_OWNER$.culture = {
				has_cultural_parameter = minimum_prowess_for_knights
			}
		}
		prowess >= 12
	}
}
