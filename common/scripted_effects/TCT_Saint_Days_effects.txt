

TCT_saint_days_completed_log_entry_effect = {
	scope:activity = {
		add_activity_log_entry = {
			key = TCT_saint_days_completed_log
			tags = { completed }
			# this line below adds the entry to the Effects section of the conclusion UI
			show_in_conclusion = yes
			character = root
			custom_tooltip = TCT_activity_result_gained_piety
			custom_tooltip = TCT_activity_result_gained_prestige
			custom_tooltip = TCT_activity_result_gained_dynasty_prestige
			show_as_tooltip = {
				every_attending_character = {
					limit = { NOT = { this = scope:host } }
					custom = EVERY_ACTIVITY_PARTICIPANT_EFFECT
					custom_tooltip = TCT_activity_result_gained_piety
					custom_tooltip = TCT_activity_result_gained_prestige
				}
			}
		}
	}
}
